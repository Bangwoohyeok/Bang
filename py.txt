import tkinter
window = tkinter.Tk()
window.title('Bang')
canvas = Canvas(window, height = 500, width = 500)
canvas.bind("<Button-1>", mouseClick);
canvas.bind("<ButtonRelease-1>", mouseDrop);

def mouseClick(event):
    global x1,y1
    x1 = event.x
    y1 = event.y

def mouseDrop(event):
    global x1,y1
    x2 = event.x
    y2 = event.y
    canvas,creat_line(x1,y1,width = 10, fill ='green')

canvas.pack();
window.mainloop();
